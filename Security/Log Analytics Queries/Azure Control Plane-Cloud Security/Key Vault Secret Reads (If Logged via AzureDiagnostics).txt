AzureDiagnostics
| where TimeGenerated >= ago(24h)
| where ResourceType =~ "VAULTS"
| where OperationName has_any ("SecretGet","GetSecret","SecretsGet")
| summarize Reads=count(), IPs=makeset(CallerIPAddress, 10) by VaultName=Resource, Identity=tostring(identity_claim_appid_g)
| order by Reads desc
